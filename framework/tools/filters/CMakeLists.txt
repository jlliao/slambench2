  message( FATAL_ERROR "SLAMBENCH_INCLUDE_DIR not found")
endif()

target_link_libraries(identity-flow-filter PRIVATE ${SLAMBENCH_C_WRAPPER} )
target_include_directories(identity-flow-filter PRIVATE ${SLAMBENCH_INCLUDE_DIR})
